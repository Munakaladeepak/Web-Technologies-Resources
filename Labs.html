<?php
function mymin($arr)
{
$n=count($arr);
$m=$arr[0];
for($i=1;$i<$n;$i++)
if($m>$arr[$i])
$m=$arr[$i];
return $m;
}
$arr=array(1,10,20,30,65);
echo (mymin($arr));
